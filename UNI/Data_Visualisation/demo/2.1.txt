        d3.select("body")
            .selectAll("p")
            .data(dataset)
            .enter()
            .append("p")
            .text(function(d) {
                return `John has watched ${d} videos today`;
            })
            .style("color", function(d) {
                if (d > 10) {
                    return "red";
                }
            });
